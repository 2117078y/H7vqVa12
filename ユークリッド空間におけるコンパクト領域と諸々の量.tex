\begin{prop}$M$ を$\mathbb R^n$ の, 境界が滑らかなコンパクト領域とする. 
\begin{align*} \vol (M) = \frac{1}{n} \int_{\partial M} (x, \nu(x)) d \mathcal H^{n -1} (x)  \end{align*}
\end{prop}
\begin{pf*}発散定理より, 
\begin{align*} \int_{\partial M} (x, \nu(x)) d \mathcal H^{n -1} (x)  = \int_{M} \div x d \vol(x) \end{align*}
が成り立ち, $\div x = \sum \partial_i x = n$ であることから従う. 
\qed
\end{pf*}


\begin{prop}(steiner公式). $A$ を滑らかな凸コンパクト領域とする. $\veps > 0$ で, $r \in [0, \veps)$ に対して$\vol(A_r)$ は$r$ の$n$ 次多項式で表される. 
\begin{align*} \vol (A_r) \coloneqq \sum_{i = 0}^ n \omega_{n-i} V_{i}(A) r^{n-i} \end{align*}

\end{prop}
\begin{pf*}

\qed
\end{pf*}
