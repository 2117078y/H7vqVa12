\section{}

\begin{prop}$(X, d)$ を距離空間とする. $x, y, z \in X$ で 
\begin{align*} d(x, y) = d(y, z) = \frac{1}{2 } d(x, z) \end{align*}
を満たすものが存在するならば, 任意の$p > 2$ に対して, 適当な$s(p) > 0$ で, $(X, s(p)d)$ が$p$ 正定値でないものが存在する. 

\end{prop}
\begin{pf*}$d(x, y) = d(y, z) = \frac{1}{2 } d(x, z) = \alpha $ と表すことにする. $\lambda = - \delta_x +  -\delta_z + 2 \delta_y$ とする. 
\begin{align*} \sum \sum \lambda(x) \lambda(y) d^p (x, y) &= -4 \alpha ^p - 4 \alpha ^p + 2(2 \alpha) ^ p \\& = 2 \alpha^p (2 ^p - 4)  \end{align*}
であるので, $p > 2$ のとき, $d^p$ は条件付き負の対称連続関数ではない. 

\qed
\end{pf*}

\begin{prop}$(X, d)$ が安定的$2$正定値距離空間であるならば, ヒルベルト空間への等長埋め込みが存在する. 

\end{prop}

\qed
