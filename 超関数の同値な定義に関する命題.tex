\title{超関数の同値な定義に関する命題}
\date{}


\author{}


\begin{document}


\maketitle

\section{}

\begin{setting}$U \subset \mathbb R^n$ を開集合とし, $\mathcal D^\prime = \mathcal D^\prime(U)$ とする. 

\end{setting}

\begin{prop}$T \in \mathcal D^\prime $ であるならば, 任意のコンパクト集合$K \subset U$ に対して, $C, m$ で
\begin{align*} \abs{T \varphi} \leq C \sum_{\abs \alpha \leq m} \sup \abs{ \partial^\alpha \varphi} \quad ( \any \varphi \in \mathcal C_c^\infty (K))  \end{align*}
を満たすものが存在する. 
\end{prop}
\begin{pf*}存在しないと仮定する(背理法). コンパクト集合$K \subset U$ で, 任意の$n \in \mathbb N $ に対して$\varphi _n \i C_c^\infty (K) $ で
\begin{align*} \abs{T \varphi _n} > n \sum_{\abs \alpha \leq n} \sup \abs{ \partial ^\alpha \varphi_n} \end{align*}
を満たすものがとれる. $c_n \in \mathbb R$ を$\abs{Tc_n\varphi_n} = 1$ となるようにとる. すると, 
\begin{align*} 1 = \abs{T c_n \varphi _n} > n \sum_{\abs \alpha \leq n} \sup \abs{ \partial ^\alpha  c_n \varphi_n}        \end{align*}
が成り立つので, 
\begin{align*} 1/n > \sum_{\abs \alpha \leq n} \sup \abs{ \partial ^\alpha  c_n \varphi_n}   \end{align*}
である. 従って, $C_c^\infty (K)$ の位相で$c_n \varphi_n \rightarrow 0 $であるが, $Tc_n \varphi_n \rightarrow 0$ ではないので, 矛盾する.
\qed
\end{pf*}

\begin{remark}
逆は明らかに成り立つので, これを同値な定義として採用してもよい. 
\end{remark}
