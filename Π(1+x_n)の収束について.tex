\date{}


\author{}

\section{}

\begin{prop} 任意の$x$ に対して
\begin{align*} 1 + x \leq e^x\end{align*}
\end{prop}
\begin{pf*}
$f(x) \coloneqq e^x - x- 1$ は$f(0) = 0$ で, 変曲点が$x = 0$ だけ. $x \geq 0 \naraba f^\prime(x) \leq 0$ なので, グラフの形は, $x = 0$ あではだんだん下がってきて, $x = 0$ で最小値$0$ をとって, そっから上昇していく. 
\qed
\end{pf*}



\begin{prop} 実数の列$\cbra{x_n}$ は$\sum x_n < \infty$ を満たすならば, 
\begin{align*} \prod (1 + x_n) < \infty \end{align*}
が成り立つ. 
\end{prop}
\begin{pf*}
\begin{align*} \prod (1 + x_n) \leq \prod e^{x_n} = e^{\sum x_n} \end{align*}
\qed
\end{pf*}
