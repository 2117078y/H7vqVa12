\section{}

\begin{setting}$mm$ 空間$(X, d, \mu)$ として十分にレギュラーなものを考えることにする. 即ち, $(X, d)$ は完備可分距離空間であり, $\mu$ は確率測度であり, $\supp \mu = X$ を仮定する. 

\end{setting}

\begin{dfn}(部分直径). $(X, d, \mu)$ を$mm$空間とする. 
\begin{align*} \diam(\mu; 1- \kappa) \coloneqq \inf \cbra{\diam(A) \mid A \in \mathcal B(X),  1 - \kappa \leq \mu(A) \leq 1} \end{align*}
と定め, これを$X$ の$\kappa$ 部分直径という. 

\end{dfn}

\begin{dfn}(オブザーバブル直径). $(X, d, \mu)$ を$mm$ 空間とする.  
\begin{align*} \textrm{ObsDiam} (X; -\kappa ) \coloneqq \sup \cbra{ \diam(f_{\#} \mu ; 1 - \kappa ) \mid f: X \rightarrow \mathbb R, 1\hyphen \textrm{Lipschitz}}\end{align*}
と定め, これを$X$ の$\kappa$ オブザーバブル直径という. 
\end{dfn}
