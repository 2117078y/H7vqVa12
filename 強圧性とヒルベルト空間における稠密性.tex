\section{}

\begin{prop}$H$ をヒルベルト空間とする. $T: H \rightarrow H$ を有界作用素とする. $c > 0$ で

\begin{align*} (Tu, u) \geq c \norm{u}^2 \quad (u \in H) \end{align*}

を満たすならば, $TH$ は$H$ で稠密である. 

\end{prop}
\begin{pf*}$(TH)^\perp = \cbra{0}$ でないと仮定(背理法). $v \in H, v \neq 0 $ で
\begin{align*} (Tu, v) = 0 \quad ( u \in H )\end{align*}
を満たすものをとると, 

\begin{align*} c \norm{v^2} \leq (Tv, v) = 0\end{align*}

より$v = 0$ となるので矛盾する. 

\qed
\end{pf*}

\begin{prop}$H$ をヒルベルト空間とする. $T: H \rightarrow H$ を有界作用素とする. $c > 0$ で

\begin{align*} (Tu, u) \geq c \norm{u}^2 \quad (u \in H) \end{align*}

を満たすならば, $T$ は全射である. 

\end{prop}
\begin{pf*}$u \in H$ を適当にとる. $v_i \in H$ で$Tv_i \rightarrow u$ を満たすものがとれる. 

\begin{align*} c \norm{v_i - v_j} ^2 \leq \norm{T(v_i - v_j)} \norm{v_i - v_j}\end{align*} 
なので, 

\begin{align*} \norm{v_i - v_j} \leq \frac{1}{c} \norm{T(v_i- v_j)}  \rightarrow 0 \end{align*}
より, $v_i$ は$H$ のコーシー列なので, 適当な$v  \in H $に収束する. 

したがって, $Tv_i \rightarrow Tv$ なので$Tv = u$ となり主張が従う.  

\qed
\end{pf*}
