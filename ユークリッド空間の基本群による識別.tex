\title{$\mathbb R^2$ と$\mathbb R^n$の基本群による識別}
\date{}


\author{}


\begin{document}


\maketitle

\section{}

\begin{prop}
$n \neq 2$ ならば, $\mathbb R^2$ は$\mathbb R^n$ と同相ではない. 
\end{prop}
\begin{pf*}
\begin{align*} \varphi^\prime  : \mathbb R^2 \rightarrow \mathbb R^n \end{align*}
という同相写像が存在すると仮定する(背理法). 
\begin{align*} \tau : \mathbb R^n \rightarrow \mathbb R^n; x \mapsto x - \varphi^\prime(0) \end{align*}
により平行移動である同相写像を定め, 
\begin{align*} \varphi = \tau \circ \varphi^\prime \end{align*}
という同相写像を定める. ($\varphi(0) = 0$ となっていることに注意する.)
\begin{align*} \varphi|_{\mathbb R^2 \setminus \cbra{0} } : \mathbb R^2 \setminus \cbra{0} \rightarrow  \mathbb R^n \setminus \cbra{0}\end{align*}
という同相があるので, 
\begin{align*} & \pi_1 (S^1) \simeq \pi_1 (S^1 ) \times \pi_1(\mathbb R) \simeq \pi_1(S^1 \times \mathbb R) \simeq \pi_1(\mathbb R^2 \setminus \cbra{0}) \\ & \quad \simeq \pi_1 (\mathbb R^n \setminus \cbra{0} )  
\simeq \pi_1(S^{n-1} \times \mathbb R) \simeq \pi_1(S^{n-1})  \times \pi_1(\mathbb R) \simeq \pi_1(S^{n-1})
\end{align*}
が成り立つ. $n \neq 2$ のときにはこれは矛盾である. 
\qed
\end{pf*}
