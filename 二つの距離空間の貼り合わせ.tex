\section{}


\begin{prop}$(X, d_X), (Y, d_Y)$ を距離空間, $\mathcal R$ を対応集合とする. $d: X \sqcup Y \times X \sqcup Y \rightarrow \mathbb R_{\geq 0}$ を$d|_{X \times X} = d_X, d|_{Y \times Y} = d_Y$ とし, $x \in X, y \in Y$ に対しては 
\begin{align*} d(x, y) \coloneqq  \inf \cbra{xx^\prime  + y^\prime y \mid (x^\prime , y^\prime ) \in  \mathcal R }\end{align*} 
 と定めると, これは距離になる. 
\end{prop}
\begin{pf*}三角不等式だけ確認することにすると, 任意に$x \in X, y \in Y$ をとり, 任意に $z \in X$ をとると, 
\begin{align*} xy &= \inf \cbra{xx^\prime + y^\prime y \mid (x^\prime, y^\prime ) \in \mathcal R} \\& \leq \inf \cbra{xz + zx^\prime + y^\prime y \mid (x ^\prime , y^\prime ) \in \mathcal R} \\&= xz + zy \end{align*}
が成り立つ. 
$z \in Y$ の時も同様である. 
\qed
\end{pf*}


\begin{prop}$(X, d_X), (Y, d_Y)$ を距離空間, $\mathcal R$ を対応集合とし, $r >0$ とする. $d: X \sqcup Y \times X \sqcup Y \rightarrow \mathbb R_{\geq 0}$ を$d|_{X \times X} = d_X, d|_{Y \times Y} = d_Y$ とし, $x \in X, y \in Y$ に対しては 
\begin{align*} d(x, y) \coloneqq  \inf \cbra{xx^\prime  + y^\prime y + r \mid (x^\prime , y^\prime ) \in  \mathcal R }\end{align*} 
 と定めると, これは距離になり, $d_H (X, Y) \leq r$ が成り立つ. 
\end{prop}
\begin{pf*} 任意の$x \in X$ に対して, $y \in Y$ で$(x, y) \in \mathcal R$ を満たすものをとると,  
\begin{align*} d(x, y) \coloneqq  \inf \cbra{xx^\prime  + y^\prime y + r \mid (x^\prime , y^\prime ) \in  \mathcal R } \leq xx + yy + r = r  \end{align*} 
が成り立つ. 距離になるかどうかは要確認. 
\qed
\end{pf*}
