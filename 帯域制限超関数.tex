\begin{dfn}(帯域制限超関数). 緩増加超関数$T \in \mathcal S^\prime (\mathbb R^n)$ で, $FT$ がコンパクトな台をもつものを, 帯域制限超関数という. 

\end{dfn}

\begin{prop}(帯域制限超関数の表現). $T$ を帯域制限超関数とする. このとき, 緩増加関数$f: \mathbb R^n \rightarrow \mathbb R$ で, 
\begin{align*} T\varphi = \int f(x)\varphi(x) dx \end{align*}
を満たすものが存在する. 
\end{prop}
\begin{pf*} $\chi \in \mathcal S$ を$\supp FT$ 上で$1$ であるものとする. 任意の$\varphi \in \mathcal S$ に対して

\begin{align*} (T, \varphi) &= (FT, F^{-1} \varphi) \\&= (\chi FT, F^{-1}\varphi) \\&= (F^{-1}(\chi FT), \varphi )   \end{align*} 

が成り立つので, 超関数の意味で

\begin{align*} T = F^{-1}(\chi FT) = F^{-1}(F(F^{-1}\chi) \cdot FT) = F^{-1} ( (2\pi)^{n/2} F((F^{-1}\chi) * T)) = (2\pi)^{n/2} F^{-1} \chi * T \end{align*}
急減少関数と, 緩増加超関数の畳み込みは$C^\infty$ 級関数であるので, 主張が従う. 
\qed
\end{pf*}
