\section{}

\begin{dfn}$S$ を集合とする. $G \subset 2^S$ が, 有限個の共通部分に関して閉じた集合族であるとき, FI集合族という. 

\end{dfn}

\begin{prop}$S$ を集合とし, $ G$ を空集合を含むFI集合族とする.  
\begin{align*} L \coloneqq \cbra{\bigcup_{i = 1}^N A_i \mid N \in \mathbb N, A_i \in G} \end{align*}
は束である. 
\end{prop}
\begin{pf*}$\bigcup_{i = 1}^N A_i , \bigcup_{i = 1}^M B_i $ に対して, $(\bigcup_{i = 1}^N A_i) \cup (\bigcup_{i = 1}^M B_i )$ は明らかに$L$ に属するので, 合併に関して閉じている. また, $(\bigcup_{i = 1}^N A_i) \cap ( \bigcup_{i = 1}^M B_i) = \bigcup_{i, j}(A_i \cap B_j)$ を考えると, $A_i \cap B_j \in G$ であるので, $L$ 属するので, 共通部分に関して閉じている. 

\qed
\end{pf*}

\begin{dfn}$S$ を集合とし, $G$ を空集合を含むFI集合族とする. 
\begin{align*} [G] \coloneqq \cbra{\bigcup_{i = 1}^N A_i \mid N \in \mathbb N, A_i \in G} \end{align*}
と定め, これを$G$ の生成する束という. 
\end{dfn}
