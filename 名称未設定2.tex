\begin{dfn}(Universal space). 距離空間$\mathcal X$ は, 任意の可分な

\end{dfn}

\begin{setting}
実数の数列全体の空間に$\sup$ ノルムを備えたものを$l^\infty$ で表すことにする. 
\end{setting}


\begin{prop}
$X$ を有界かつ可分な距離空間とする. $\cbra{w_n}$を$X$ の可算稠密部分集合とする.
\begin{align*} \iota : x \mapsto ( d(w_1, x) , d(w_2, x) , \ldots    )\end{align*}
により$\iota: X \rightarrow l^\infty$ を定めると, これは等長写像である. 
\end{prop}
\begin{pf*}
\begin{claim}
\begin{align*} \sup_n \abs{d(x, w_n) - d(y, w_n)} \leq d(x, y)    \end{align*}
が成り立つ. 
\end{claim}
\begin{claimproof}
任意の$n$ に対して
\begin{align*} d(x, w_n) \leq d(x, y) + d(y, w_n), \quad d(y, w_n) \leq d(x, y) + d(x, w_n),    \end{align*}
\end{claimproof}

また

\begin{claim}
$\sup_n \abs{d(x, w_n) - d(y, w_n)} \geq d(x, y)$
\end{claim}
\begin{claimproof}
\begin{align*} d(x, y) - 2 d(x, w_n) \leq d(x, w_n) + d'(w_n , y) - 2 d(x, w_n) \leq \abs{d(x, w_n) - d(y, w_n)} \end{align*}
が成り立つ. $\cbra{w_n]$ は稠密なので, 

\begin{align*}\sup \cbra{\abs{d(x, w_n) - d(y, w_n)}  } \geq \sup  d(x, y) - 2 d(x, w_n) = d(x, y) - 2 \inf d(x, w_n) = d(x, y) \end{align*}
が成り立つ.
\end{claimproof}

よって示された. 

\qed
\end{pf*}
