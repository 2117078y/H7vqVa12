\begin{dfn}急減少関数全体$\mathcal S$ に
\begin{align*} \norm \varphi _N \coloneqq \sum_{\abs \alpha, \abs \beta < N} \norm{x}^\alpha \abs{\partial_x^\beta f(x) }  \end{align*}
によりセミノルムを定める. 
\end{dfn}

\begin{prop}$\varphi \in \mathcal S$ に対して, 
\begin{align*} \norm \varphi_N  \geq \norm x ^N \sup  \abs{f(x)}  \end{align*}
\end{prop}
\begin{pf*}
あたりまえ.
\qed
\end{pf*}


\begin{prop}$f: \mathbb R^n \rightarrow \mathbb R$ を局所可積分関数とする. $M \in \mathbb N$ で
\begin{align*} \int_{\norm x \geq 1} \frac{\abs{f(x)}}{ \norm{x} ^M} < \infty \end{align*}
をみたすものが存在するならば, $f$ は緩増加超関数を定める. 

\end{prop}
\begin{pf*}
任意に$\varphi \in \mathcal S$ をとる. 適当な$C>0$ で
\begin{align*}  \abs{(f, \varphi)} \leq \int_{\norm x <  1} \abs{f(x)}\abs{ \varphi(x)}dx + \int_{\norm x \geq  1} \frac{\abs{f(x)}}{ \norm x^M} \norm{x} ^M \abs{ \varphi(x)} \leq C \norm \varphi _M    \end{align*}
を満たすものが存在する. $\mathcal S$ における収束$\varphi_n \rightarrow \varphi$ は, 任意のセミノルムに関しての収束であるので, もちろん$M \in \mathbb N$ に関しても収束する.  
\qed
\end{pf*}
