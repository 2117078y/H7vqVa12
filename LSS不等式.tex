\begin{notation}$(X, d)$ を距離空間とする. $x, y \in X$ に対して
\begin{align*} xy \coloneqq d(x, y)\end{align*}
という記号を用いることにする. 
\end{notation}

\begin{dfn}$(X, d)$ を距離空間とする. $n + 1$ 点$\cbra{ p, x_1, \ldots, x_n}$ に対して, 
\begin{align*} k_p (x_i, x_j) \coloneqq x_i p ^2 + x_j p^ 2 - x_i x_j ^2 \end{align*}
と定める. 任意の非負関数$\lambda : \cbra{ p, x_1, \ldots, x_n} \rightarrow \mathbb R$ に対して
\begin{align*} \sum_i \sum_j \lambda(x_i) \lambda (x_j) k_p (x_i, x_j) \geq 0 \end{align*}
が成り立つとき, $\cbra{ p, x_1, \ldots, x_n}$ は
$p$ を中心とする($n$ 点)Lang-Schroeder-Strum不等式を満たすと言う. 

\end{dfn}

\begin{prop}$\cbra{p, x_1, \ldots, x_n}$ 上で$d^2$ が条件付き負であるならば, $\cbra{p, x_1, \ldots, x_n}$ は$p$ 中心LNN不等式を満たす. 

\end{prop}
\begin{pf*}
対称連続関数のpdfを参照. 
\qed
\end{pf*}

\begin{remark}LSS不等式は
「任意の非負関数$\lambda : \cbra{ p, x_1, \ldots, x_n} \rightarrow \mathbb R$ に対して」という条件であるので, 対称連続関数が正であるという性質よりは弱い. 従って, 逆は成り立たない. 
\end{remark}
