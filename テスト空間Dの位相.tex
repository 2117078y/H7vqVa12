\section{}

\begin{setting}

\begin{align*} C_c^\infty (k) \coloneqq \cbra{\varphi \in C_c^\infty (\mathbb R^n) \mid \supp \varphi \subset k}  \end{align*}

と定める. 
$\Omega \subset \mathbb R^n$ を開集合とする. $k \Subset \Omega$ と非負整数$m$ に対して

\begin{align*} p_{m,k}(\varphi) \coloneqq \sup_{x \in k} \sum_{\abs \alpha \leq m} \abs{\partial^\alpha \varphi (x)}\end{align*}

というセミノルムにより位相を定める. 

\end{setting}

\begin{dfn}(超関数$\mathcal D^\prime (\Omega)$). $\Omega \subset \mathbb R^n$ を開集合とする. $C_c^\infty(\Omega )$ 上の線型形式$T$ で, 任意の$k \Subset \Omega $ に対して非負整数$m(k)$ と定数$c_k$ で
\begin{align*} \abs{T\varphi} \leq c_k p_{m(k), k} (\varphi) \quad(\varphi \in C_c^\infty(k))\end{align*}

を満たすものが存在する時に, $T$ を$\Omega $ 上の超関数という. これら全体を$\mathcal D^\prime (\Omega)$ により表す. 

\end{dfn}

\begin{dfn}$\mathcal D^\prime (\Omega)$ における連続性を, $\varphi_i \rightarrow \varphi$ を, 適当な$k \Subset \Omega $ で, $\supp \varphi_i \subset k \quad (\any i)$  かつ
\begin{align*} p_{m,k}(\varphi_i - \varphi) \rightarrow 0 \quad (\any m ) \end{align*}
を満たすものが存在すること. 
\end{dfn}

\begin{remark}
つまり, 適当な相対コンパクト集合を取ればその上でセミノルム収束するということ. 
\end{remark}
