\section{}

\begin{prop}$2$点距離空間$(\cbra{x, y}, d)$のマグニチュードは
\begin{align*} \Mag^p(\cbra{x, y}) \coloneqq 1 + \tanh \frac{d^p(x, y)}{2} \end{align*}
で与えられる. 
\end{prop}
\begin{pf*}
計算すればわかる. 
\qed
\end{pf*}


\begin{prop}$X = (\cbra{x, y}, d)$ を2点距離空間とする. $0 < p < q < \infty $ とすると, \\
(1)$0 < s < \frac{1}{D} \naraba \Mag^p (sX) > \Mag^q (sX) .$ \\
(2)$s = \frac{1}{D} \naraba \Mag^p (sX) = \Mag^q (sX).$ \\
(3)$\frac{1}{D} < s < \infty \naraba \Mag^p (sX) < \Mag^q (sX).$ \\
が成り立つ. ただし, $D \coloneqq d(x, y)$ と定めてある. 

\end{prop}
\begin{pf*}

\begin{align*} \paren{1 + \tanh \frac{D^q}{2}   } -  \paren{1 + \tanh \frac{D^p}{2}} =  \tanh \frac{D^q}{2}  -  \tanh \frac{D^p}{2}  \end{align*}
である. $\tanh $ が単調増大であるので, 
\begin{align*} \frac{D^p}{2} \leq \frac{D^q}{2}\end{align*} 
となるのはいつかを考えれば良い. 
\qed
\end{pf*}

\begin{prop}$X = (\cbra{x, y}, d)$ を2点距離空間, $0 < p < \infty$ とする. \\
(1)$ \lim_{p\rightarrow 0} \Mag^p (X) = 1.$ \\
(2)$ \lim_{p\rightarrow \infty} \Mag^p (X) = 2. $\\
が成り立つ.
\end{prop}
\begin{pf*}
明らかである. 
\qed
\end{pf*}
