\section{}

\begin{prop}$K \subset \mathbb R^n$ を原点を含む対称凸体とする. このとき, $\norm{\cdot }_K : \mathbb R^n \rightarrow \mathbb R_{\geq 0} $ を
\begin{align*} \norm {x} _K \coloneqq \min \cbra{t \geq  0 \mid x \in tK}\end{align*}
により定めると, これはノルムである. 
\end{prop}
\begin{pf*}三角不等式以外は明らかである. 任意の$x, y \in \mathbb R^n$ に対して
\begin{align*} \frac{x}{\norm x_K + \norm y_K} \in K, \quad \frac{y}{\norm x_K + \norm y_K} \in K \end{align*}
であることがわかるので, 
\begin{align*} \frac{x + y}{\norm x_K + \norm y_K} \in K\end{align*}
が従う. 故に
\begin{align*} \norm{x + y}_K \leq \norm x_K + \norm y_K   \end{align*}
が成り立つ. 
\qed
\end{pf*}

\begin{prop}$K \subset \mathbb R^n$ を原点を含む対称凸体とする. $B(0; R) \subset K$ が成り立つならば, 任意の$x \in K$ に対して
\begin{align*} \norm x_K \leq \frac{1}{R} \norm{x}_2 \end{align*}
が成り立つ. 
\end{prop}
\begin{pf*}
\begin{align*} \frac{x}{\norm x / R} = R \frac{x}{\norm x} \in \bar B (0; R) \subset K\end{align*}
が成り立つので, 主張が従う. 
\qed
\end{pf*}

\begin{prop}$K \subset \mathbb R^n$ を原点を含む対称凸体とする. $B(0; R) \subset K$ が成り立つならば, 写像$x \mapsto \norm x _K$ は$\frac{1}{R} \hyphen $ Lipschitz 写像である. 

\end{prop}
\begin{pf*}前述の命題より, 任意$x, y \in K$ に対して, 
\begin{align*} \abs{\norm{x}_K - \norm y _K} \leq \norm{x - y}_K  \leq \frac{1}{R}\norm{x - y}_2  \end{align*}
が成り立つ. 
\qed
\end{pf*}
