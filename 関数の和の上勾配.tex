\begin{prop}(関数の和の上勾配). $(X, d)$ を距離空間とする. $u, u_1, u_2: X \rightarrow \overline{\mathbb R}$ を拡張実数値関数とする. $g \in UG(u), g_1 \in UG(u_1), g_2 \in UG(u_2)$ に対して, 
\begin{align*} &g_1 + g_2 \in UG(u_1 + u_2) \\ &\abs a g \in UG(au) \quad (a \in \mathbb R )  \end{align*} 
が成り立つ. 
\end{prop}
\begin{pf*}省略. 

\qed
\end{pf*}


\begin{remark}$u_1(x) \coloneqq x , u_2(x) \coloneqq - x$ とすると, $1 \in UG(u_1) , 1 \in UG(u_2)$ であるわけだが, 
\begin{align*} u_1(x) - u_2(x) = 2x  \end{align*}
であるので, $1 - 1 = 0 $ は$UG(u_1 - u_2)$ に属さない. 

\end{remark}
