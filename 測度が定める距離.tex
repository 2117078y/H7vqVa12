\date{}

\section{}

\begin{prop}$X$ を距離空間, $\mu$ を$X$ 上の外測度, $S: X \rightarrow X \rightarrow \mathcal P (X) $ とする. \\
(1)$S(x,y) = S(y, x) \quad (\any x, y \in X)$ \\
(2)$\mu (S (x, y)) = 0 \LR x = y$ \\ 
(3)$S(x, y) \subset S(x, z) \cup S(z, y) \quad (x, y, z \in X)$ \\
を満たすとする. このとき, 
\begin{align*} d(x, y) \coloneqq \mu(S(x, y)) \end{align*}
は$X$ 上の距離を定める. 
\end{prop}
\begin{pf*}
適当に確かめたらわかる. 
\qed
\end{pf*}

\begin{dfn}外測度と集合値関数の組($\mu, S$) が前述の命題の$(1), (2), (3)$ を満たすとき, 距離を誘導するという. 

\end{dfn}


\begin{ex}実数直線$\mathbb R$ において, 
\begin{align*} S(x, y) \coloneqq [x, y] \end{align*}
と定める. $\mu$ としてルベーグ測度をとる. 
\end{ex}




\begin{ex}平面の中の三脚$T \subset \mathbb R^2$ を考える. 
\begin{align*} S(x, y) \coloneqq x\textrm{から}y \textrm{への最短経路}\end{align*}
と定める. $\mu$ としてルベーグ測度をとる. 
\end{ex}
