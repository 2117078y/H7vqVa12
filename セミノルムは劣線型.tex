\section{}




\begin{dfn}(セミノルム). $V$ を$\mathbb K = \mathbb R \,\, (\textrm{or} \,\, \mathbb C)$ 上のベクトル空間とする. $V$ 上の非負実数値間数
\begin{align*} \norm \cdot : V \rightarrow \mathbb R _{\geq 0} \end{align*} 
で\\
(1)$\norm{v + w} \leq \norm v + \norm w \quad (v,w \in V) $ \\
(2)$\norm{kv} = \abs k \norm v \quad (v \in V, k \in \mathbb K)$ \\
を満たすものをセミノルムという. 
\end{dfn}

\begin{dfn}(劣線型関数). $V$ を$\mathbb R $ 上のベクトル空間とする. $V$ 上の実数値関数
\begin{align*} p : X \rightarrow \mathbb R  \end{align*} 
で\\
(1)$ p(v + w) \leq p( v) + p( w) \quad (v,w \in V) $ \\
(2)$p (rv) = r p( v) \quad (v \in V, r \in \mathbb R _{\geq 0})$ \\
を満たすものを劣線型関数という. 

\end{dfn}

\begin{prop}
$V$ を$\mathbb R$ 上のベクトル空間とする. $V$ 上のセミノルムは, 劣線型関数である. 
\end{prop}
\begin{pf*}

明らか.


\qed
\end{pf*}
