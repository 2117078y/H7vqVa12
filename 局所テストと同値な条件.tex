\section{}

\begin{setting}
$\mathcal D^\prime = \mathcal D^\prime (\mathbb R^n)$ とする. 
\end{setting}

\begin{dfn}$T \in \mathcal D^\prime $ が$x \in \mathbb R^n$ において局所的に$C^\infty$ であるとは, $x$ の開近傍$U_x$ と$f \in C^\infty (\mathbb R^n)$ で, $\supp \varphi \subset U_x$ である$\varphi \in \mathcal D$ に対して 
\begin{align*} T\varphi = \int f(x) \varphi (x) dx \end{align*}
を満たすものが存在することをいう. 
\end{dfn}

\begin{remark} こういう$f$ を, $T$ を$U_x$ において代表する関数ということにする.   

\end{remark}


\begin{prop}$T \in \mathcal D^\prime $ が$x \in \mathbb R^n$ において局所的に$C^\infty$ であることと, $\chi (x) \neq 0 $ である$\chi \in C_c^\infty (\mathbb R^n) $ に対して$\chi T \in C_c^\infty (\mathbb R^n)$ が成り立つことは必要十分である. 

\end{prop}
\begin{pf*}$(\gyaku). $ 明らか. $(\naraba)$.  適当な開集合$U_x$ と, $T$ を$U_x$ において代表する関数$f \in C^\infty(U_x)$ をとる. $\chi \in C_c^\infty (\mathbb R^n) $ として, $x$ で局所的に$\chi(x) \neq 0 $ であるようなものをとると,  
任意の$\varphi \in \mathcal D ^\prime $ に対して, 

\begin{align*} (\chi T, \varphi) = (T, \chi \varphi)  \end{align*} 

であるが, $\supp \chi \varphi \subset U_x $ であるので, 

\begin{align*} (T, \chi \varphi ) = \int f(x) \chi(x) \varphi(x) dx \end{align*}
が成り立つので, 超関数の意味で$\chi T = \chi f$ であるが, 右辺が$C_c^\infty(\mathbb R^n) $ であるので, 主張が従う. 
\qed
\end{pf*}
