\begin{dfn}$(X, d)$ を距離空間, $f$ を$X$ 上の局所リプシッツ関数とする. 任意の$X$ の速度$1$ の絶対連続曲線$\gamma: [0, L\gamma] \rightarrow \mathbb X$ に対して
\begin{align*} \abs{f(\gamma(L\gamma)) - f(\gamma(0)) }= \int_0^{L\gamma} g(\gamma(t)) dt  \end{align*}
を満たすボレル関数$g: X \rightarrow \mathbb R$ を, $f$ の上勾配という. $f$ の上勾配全体を$UG(f)$ で表すことにする. 

\end{dfn}

\begin{dfn}(ニュートン空間). $(X, d)$ を距離空間とする. 
\begin{align*} N^{1, p} \coloneqq \cbra{u \in  L^p(X) \mid \paren{ \int_{X} \abs u ^p d \mu + \inf _{g \in UG(u)}  \int_X g^p d\mu } < \infty }\end{align*}
と定め, これをニュートン空間という. 
\end{dfn}
